function sendMove(move) { fetch("/api/game", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ move: move }) }) .then(response => response.json()) .then(data => { window.location.href = "/game.html?move=" + move + "&result=" + data.result + "&computerMove=" + data.computerMove; }) .catch(error => { console.error("Error:", error); }); }
